    def undoable(&block)
      @undo_stack ||= []
      @open_undo_stack ||= @undo_stack
      if @open_undo_stack == @undo_stack
        (@undo_stack ||= []) << (newstack=[])
        @open_undo_stack = newstack
        yield
        @open_undo_stack = @undo_stack
      else
        yield
      end
    end
