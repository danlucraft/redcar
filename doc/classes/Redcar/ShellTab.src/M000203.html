    def down
      @history_point ||= 0
      if @history_point < -1
        @history_point += 1
        delete(@start_loc, contents.length)
        insert(@start_loc, @history[@history_point])
      elsif @history_point == -1
        @history_point += 1
        delete(@start_loc, contents.length)
        insert(@start_loc, @current)
        self.cursor = @start_loc + @current.length
      end
    end
