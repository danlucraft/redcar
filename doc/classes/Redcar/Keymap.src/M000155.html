    def Keymap.find_all_keymaps(keybinding)
      keymaps = []
      ObjectSpace.each_object(Redcar::Keymap) do |obj|
        if obj.respond_to? :keymap_respond_to?
          if obj.keymap_respond_to? keybinding
            keymaps << obj
          end
        end
      end
      keymaps
    end
