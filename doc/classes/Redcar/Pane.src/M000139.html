    def initialize(panes, notebook, options={})
      options = process_params(options,
                               { :tab_position => :top,
                                 :tab_angle    => :horizontal })
      @panes = panes
      @notebook = notebook
      @count = 0
      @tab_hash = {}
      @tab_position = options[:tab_position]
      @tab_angle    = options[:tab_angle]
    end
