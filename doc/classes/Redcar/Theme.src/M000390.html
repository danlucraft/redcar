    def settings_for_scope(scope)
      applicables = []
      @settings.each do |setting|
        if setting['scope']
          if spec = applicable?(setting['scope'], scope)
            applicables << [spec, setting]
          end
        end
      end
      applicables.sort_by {|a| -a[0]}.map {|a| a[1]}
    end
