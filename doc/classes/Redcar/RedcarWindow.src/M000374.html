    def get_panes(hash)
      obj = []
      ["left", "right", "top", "bottom"].each do |dir|
        if hash[dir].is_a? Redcar::Pane
          obj << hash[dir]
        else
          if hash[dir]
            obj += get_panes(hash[dir])
          end
        end
      end
      obj
    end
