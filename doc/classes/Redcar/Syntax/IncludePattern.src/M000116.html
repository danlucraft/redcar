      def initialize(hash)
        case hash["include"][0..0]
        when '#'
          @type = :repository
          @value = hash["include"][1..-1]
        when '$'
          if hash["include"] == "$self"
            @type = :self
          else
            raise ArgumentError, "unknown $ include: '#{hash['include']}'"
          end
        else
          @type = :scope
          @value = hash["include"]
        end
      end
