      def get_capture(str, md)
        capture_index = str.to_i-1
        if capture_index == -1
          capture = md.to_s
        else
          tmp = md.captures.map{|c| (c==nil ? "" : c)}
          capture = tmp[capture_index]
        end
        if capture == nil
          p md
          p md.to_s
          p md.captures
          p capture_index
          raise StandardError, "capture not found"
        end
        capture
      end
