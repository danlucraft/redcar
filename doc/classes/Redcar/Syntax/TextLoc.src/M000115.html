      def <=>(other)
        if self.line < other.line
          -1
        elsif self.line > other.line
          1
        elsif self.line == other.line
          if self.offset < other.offset
            -1
          elsif self.offset > other.offset
            1
          elsif self.offset == other.offset
            0
          else
            raise ArgumentError
          end
        else
          raise ArgumentError
        end
      end
