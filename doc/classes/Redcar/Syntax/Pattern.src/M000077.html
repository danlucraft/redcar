      def initialize(hash)
        unless hash["name"] or hash["contentName"]
          hash["name"] = $spare_name.clone
          $spare_name = $spare_name.succ
        end
        @name = hash["name"] || hash["contentName"]
        @captures = hash["captures"] || {}
        @captures.each do |key, value|
          @captures[key] = value['name']
        end
      end
